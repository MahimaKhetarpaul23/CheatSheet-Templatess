ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
    if(list1==NULL){
        return list2;
    }
    
    if(list2==NULL){
        return list1;
    }
    
    ListNode* head;
    ListNode* temp1=list1;
    ListNode* temp2=list2;
    ListNode* prev=NULL;
    
    while(temp1!=NULL && temp2!=NULL){
        if(temp1->valval){
            if(prev==NULL){
                head=temp1;
                prev=temp1;
                ListNode* temp = temp1->next;
                temp1->next=NULL;
                temp1=temp;
            }
            else{
                ListNode* temp = temp1->next;
                prev->next=temp1;
                prev=temp1;
                // temp1->next=NULL;
                temp1=temp;    
            }
        }
        else{
            if(prev==NULL){
                head=temp2;
                prev=temp2;
                ListNode* temp = temp2->next;
                temp2=temp;
            }
            else{
                ListNode* temp = temp2->next;
                prev->next=temp2;
                prev=temp2;
                temp2=temp;    
            }
        }
    }            
    if(temp1!=NULL){
        if(prev==NULL){
            head=temp1;
            prev=temp1;
        }
        else{
            prev->next=temp1;
        }
    }
    
    if(temp2!=NULL){
        if(prev==NULL){
            head=temp2;
            prev=temp2;
        }
        else{
            prev->next=temp2;
        }
    }
    
    return head;   
}
Find inorder of Binary tree 
               
void inorder(TreeNode* root,vector &in){
    if(root==NULL){
        return;
    }
    inorder(root->left,in);
    in.push_back(root->val);
    inorder(root->right,in);
}

vector inorderTraversal(TreeNode* root) {
    vector v;
    inorder(root,v);
    return v;
}
Valid Parenthesis : Check if brackets are valid or not. 
               
bool isValid(string s) {
    stack stk;
    for(int i=0;i< s.size();i++){}
        if(s[i]=='(' || s[i]=='{' || s[i]=='['){
            stk.push(s[i]);
        }
        else if(s[i]==')'){
            if(stk.empty() || stk.top()!='('){
                return false;
            }
            stk.pop();
        }
        else if(s[i]=='}'){
            if(stk.empty() || stk.top()!='{'){
                return false;
            }
            stk.pop();
        }         
        else if(s[i]==']'){
            if(stk.empty() || stk.top()!='['){
                return false;
            }
            stk.pop();
        }
    }
    if(stk.empty()){
        return true;
    }
    return false;
}
